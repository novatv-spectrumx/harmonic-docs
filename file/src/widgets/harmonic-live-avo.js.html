 /**
 * The HarmonicLiveAVO widget provides control of the system's live
 * audio voice over unit (AVO).
 *
 * The HarmonicLiveAVO widget requires a single createjs.MovieClip to be
 * registered.  This widget is defined in the Harmonic Template Widget
 * Library.  It just needs to be included in the template for it to work.
 */
class HarmonicLiveAVO {

    /**
     * Constructor
     *
     * @param {Number} unit - LiveAVO unit number (0-1)
     */
    constructor(unit, enable = true) {
        hjs.debug("HarmonicLiveAVO.constructor: unit: " + unit);

        /**
         * Live AVO unit number
         * @type {Number}
         */
        this._unit = unit;

        /**
         * IsLive flag.  This will get set to true when the template
         * becomes visible.
         * @type {Boolean}
         */
        this._isLive = false;

        /**
         * Enable flag.  Controlled by the client and defaults to true.
         * @type {Boolean}
         */
        this._enable = enable;

        /**
         * The Live AVO object that is registered with this class.
         * @type {Object}
         */
        this._liveAVO = undefined;
    }

    /**
     * Register the object that represent the Live AVO object.
     * The registered object must be a MovieClip.
     *
     * @param {Object} liveAVO - createjs object that represents the liveAVO
     * @return {Boolean} - true if registration was successful, false otherwise.
     */
    register(liveAVO) {
        if (this._liveAVO === undefined) {

            hjs.debug("HarmonicLiveAVO.register: unit: " + this._unit);

            if (!(liveAVO instanceof createjs.MovieClip)) {
                hjs.error("HarmonicLiveAVO.register: LiveAVO is not a container");
                return false;
            }

            this._liveAVO = liveAVO;

            this._init();

            return true;
        }

        return false;
    }

    /**
     * After successful registration, this method will initialize everything
     * necessary for the Live AVO to function.
     *
     * @listens {HarmonicTemplate.VISIBLE_EVENT} when the template is visible
     */
    _init() {
        // Make the liveAVO object invisible on the platform
        if (hjs.isOnPlatform()) {
            this._liveAVO.visible = false;
        }

        // Listen for the VISIBLE_EVENT.  This will set the visible flag to
        // true.
        document.addEventListener(HarmonicTemplate.VISIBLE_EVENT, (event) => {
            this._isLive = true;

            hjs.debug("HarmonicLiveAVO.setLiveAVOIsLive: " + this._isLive);
            hjs.setLiveAVOIsLive(this._unit, this._isLive);
        });

        this._setEnable(this._enable);
    }

    //--------------------------------------------------------------------------
    // Private Methods
    //--------------------------------------------------------------------------

    /**
     * Sets the enable property locally and sends a websocket message to the
     * HTML renderer so that it can be updated on the shared surface.
     * @param {Boolean} enable
     */
    _setEnable(enable) {
        this._enable = enable;
        hjs.enableLiveAVO(this._unit, this._enable);
    }

    //--------------------------------------------------------------------------
    // Properties
    //--------------------------------------------------------------------------

    /**
     * Gets the unit number
     * @return {Number} the unit number associated with this Live AVO.
     */
    get unit() {
        return this._unit;
    }

    /**
     * Get the isLive flag.  This returns the local copy.
     * @return {Boolean} the current state of the cached isLive flag.
     */
    get isLive() {
        return this._isLive;
    }

    /**
     * Gets the current enable state of the live AVO.
     * @return {Boolean} true = enabled, false = disabled
     */
    get enable() {
        return this._enable;
    }

    /**
     * Sets the enable state of the live AVO.
     * @param {Boolean} value - true = enable, false = disable
     */
    set enable(value) {
        this._setEnable(value);
    }
}
