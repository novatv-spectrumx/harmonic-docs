/**
 * The HarmonicCountdown widget provides a countdown text field that can
 * be controlled with the widget controls.
 *
 * This class implements the Harmonic Countdown Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The HarmonicCountdown widget requires that the registered createdjs
 * object have the following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 * - The createjs.MovieClip must contain a createjs.Text child named 'harmonicField'.
 *   This text object represents the dynamic text field.
 *
 * This class defines functions for:
 *
 * - updateTextField - converts the text to a number and updates the countdown
 * - restartFieldAnimation - restarts the countdown
 * - stopFieldAnimation - stops the countdown
 *
 * To use this class, simply register the createjs object using the
 * static register function.  The created HarmonicCountdown object is
 * returned from the register function.  If there is a error, then
 * 'undefined' is returned and an error will be logged.
 *
 * The 'countdownFrom' can be specified in the register function or it
 * can be set using the 'countdownFrom' property on the object returned from
 * the call to register.
 *
 * When the countdown reaches zero, the widget will dispatch the
 * HarmonicCountdown.COUNTDOWN_COMPLETE_EVENT event on the document.
 *
 * The manualControl property allows the countdown to manually controlled
 * independently from the start/stop field animation controls.  This may
 * be useful for countdown timers that need to start in the middle of an
 * animation.
 *
 * @example
 * 1. When the createjs object is registered with the class in the first frame of the movie clip:
 * let countdown = HarmonicCountdown.register(this, 20, false);  // set countdown to 20 seconds with no manual control
 *
 * 2. Or set the 'countdownFrom' property from the returned object:
 * let countdown = HarmonicCountdown.register(this);
 * countdown.countdownFrom = 20;
 *
 * document.addEventListener(HarmonicCountdown.COUNTDOWN_COMPLETE_EVENT, (event) => {
 *     hjs.info("The countdown has completed");
 * });
 */
class HarmonicCountdown extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     * - adds a listener to the 'tick' event so that the count can be updated
     *
     * @listens {'tick'} the countdown is updated every tick
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     * @param {Object} includeInTemplateDescription - if true, the Harmonic Template will include in the template description
     * @param {Number} countdownFrom - number in seconds to countdown from
     * @param {Boolean} manualControl - manual control of the countdown
     */
    constructor(widgetType, fieldNum, name, createjsObj, countdownFrom, manualControl) {
        super(widgetType, fieldNum, name, createjsObj, true);

        hjs.debug("HarmonicCountdown.constructor");

        /**
         * Flag indicating if the countdown is started.
         * @type {Boolean}
         */
        this._isStarted = false;

        /**
         * Flag indicating if the countdown is manually controlled.  If manually
         * controlled, then the widget will not respond to the start and stop animation
         * events.
         * @type {Boolean}
         */
        this._manualControl = manualControl;

        /**
         * The countdownFrom property.
         * @type {Number}
         */
		this._countdownFrom = countdownFrom;

        /**
         * Current value of the countdown (in seconds).
         * @type {Number}
         */
        this._countdown = countdownFrom;

        /**
         * Milliseconds per tick event.
         * @type {Number}
         */
        this._msPerTick = 0;

        // Determine the milliseconds per tick based on the frame rate
        const fps = createjs.Ticker.framerate;
        if ((fps >= 59.8) && (fps <= 60.2)) {
            this._msPerTick = 1001/60000;
        }
        else {
            this._msPerTick = 1/fps;
        }

        // Listen for the 'tick' event on the object
        createjsObj.addEventListener("tick", (event) => {
            this._onTick();
        });
    }

    /**
     * Registers a createjs object with the widget.  A new
     * HarmonicCountdown will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with the countdown
     * @param {Number} countdownFrom - number of seconds to countdown from
     * @param {Boolean} manualControl - true = manually control the countdown
     *
     * @return {Object} the newly created HarmonicCountdown instance, or undefined if there is an error
     */
    static register(createjsObj, countdownFrom = 0, manualControl = false) {
        let result = HarmonicWidget.validate("HarmonicCountdown", createjsObj, true);
        if (result) {
            return new HarmonicCountdown(HarmonicTemplate.WIDGET_TYPE_COUNTDOWN, result.fieldNum, result.name, createjsObj, countdownFrom, manualControl);
        }
        else {
            return undefined;
        }
    }

    //--------------------------------------------------------------------------
    // Public Methods and Properties
    //--------------------------------------------------------------------------
    /**
     * Starts the countdown from the countdownFrom value.
     */
    start() {
        this._isStarted = true;
        this._countdown = this._countdownFrom;
    }

    /**
     * Resumes the countdown.
     */
    resume() {
        this._isStarted = true;
    }

    /**
     * Stops the countdown.
     */
    stop() {
        this._isStarted = false;
    }

    /**
     * Sets the countdownFrom value.
     * @param {Number} value - the number of seconds to countdown from
     */
    set countdownFrom(value) {
        this._countdownFrom = value;
    }

    /**
     * Gets the countdownFrom value.
     * @return {Number}
     */
    get countdownFrom() {
        return this._countdownFrom;
    }

    /**
     * Gets the current value of the countdown.
     * @return {Number}
     */
    get countdown() {
        return this._countdown;
    }

    /**
     * Set the manualControl value.
     * @param {Boolean} value - true = enable, false = disable
     */
    set manualControl(value) {
        this._manualControl = value;
    }

    /**
     * Gets the manualControl value.
     * @return {Boolean}
     */
    get manualControl() {
        return this._manualControl;
    }

    //--------------------------------------------------------------------------
    // Private Methods
    //--------------------------------------------------------------------------

    /**
     * Handles the updateTextField request from the Harmonic Template.
     *
     * @param {String} text - a string representation of the number to set the countdown to
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    updateTextField(text, render) {
        this._countdownFrom = parseInt(text);
        this._countdown = this._countdownFrom;
    }

    /**
     * Handles the restartFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} fromBeginning - flag indicating if the widget should restart animation from the beginning
     */
    restartFieldAnimation(fromBeginning) {
        if (this._manualControl) {
            return;
        }

        if (fromBeginning) {
            this._countdown = this._countdownFrom;
        }

        this._isStarted = true;
    }

    /**
     * Handles the stopFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} immediate - flag indicating if the widget should stop immediately or at the end of the animation
     */
    stopFieldAnimation(immediate) {
        if (this._manualControl) {
            return;
        }

        if (immediate) {
            this._isStarted = false;
        }
    }

    /**
     * 'tick' handler that updates the countdown
     *
     * @emits {HarmonicCountdown.COUNTDOWN_COMPLETE_EVENT} dispatched when the countdown to zero completes
     */
    _onTick() {
        if (this._isStarted) {
            if (this._countdown < 0) {
                this._isStarted = false;

                // Dispatch the complete event when the countdown finishes.
                let event = new CustomEvent(HarmonicCountdown.COUNTDOWN_COMPLETE_EVENT,
                    {
                        detail : this
                    }
                );

                document.dispatchEvent(event);
            }
            else {
                this._countdown -= this._msPerTick;
            }

            this._updateCountdown(this._countdown);
        }
    }

    /**
     * Updates the createjs object's text with the value from countdown.
     *
     * @param {Number} countdown - the count value to display
     */
    _updateCountdown(countdown) {
        if (countdown >= 0) {
            const hours = Math.floor(countdown / 3600);
            const minutes = Math.floor((countdown % 3600) / 60);
            const seconds = Math.floor(countdown % 60);
            let text = "";

            if (hours > 0) {
                text += hours.toString() + ":";
            }

            if (minutes == 0) {
                if (hours > 0) {
                    text += "00";
                }

                text += ":";
            }
            else if (minutes < 10) {
                if (hours > 0) {
                    text += "0" + minutes.toString() + ":";
                }
                else {
                    text += minutes + ":";
                }
            }
            else {
                text += minutes + ":";
            }

            if (seconds < 10) {
                text += "0" + seconds;
            }
            else {
                text += seconds;
            }

            this._createjsObj.harmonicField.text = text;
        }
        else {
            this._createjsObj.harmonicField.text = "";
        }
    }
}

//------------------------------------------------------------------------------
// Events
//------------------------------------------------------------------------------
HarmonicCountdown.COUNTDOWN_COMPLETE_EVENT = "CountdownCompleteEvent";
