/**
 * The HarmonicVideo widget provides a video component that is synchronized with
 * the Harmonic Template timeline.  Since loading an external video source
 * takes time, this widget will hold off load complete until the video can play
 * through without additional fetching.  This component will start the video at
 * the correct time, but playback is controlled by Chrome so frame accuracy is
 * not alway achievable.
 *
 * CAUTION:
 * - Running fullscreen video files inside a HTML template is not recommended.
 * - Video files should be as small as possible so that they do not impact the
 *   template playback.
 * - Run fullscreen video files separately on their own layer.  Not in a HTML template.
 * - Running fullscreen video files inside a HTML template may impact playback of the
 *   template and will most likely result in repeat or missed frames.
 *
 * This class implements the Harmonic Video Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The HarmonicVideo widget requires that the registered createdjs object have
 * the following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 * - The createjs.MovieClip must contain a video component with the class parameter
 *   set to a unique name.  This name is used to locate the video element in the DOM
 *   once it is created by the Adobe code.
 *
 * In order for video element to be located in the DOM, the value specified in
 * the Component Parameter class text box in Adobe Animate must be passed to the
 * register function.  This is required since there is no way to identify the
 * video element that the Adobe component code creates.
 *
 * This class defines functions for:
 *
 * - updateTextField - allows changes to 'src' and 'currentTime'
 *   - Syntax:
 *     - src:<name of src>
 *     - currentTime:<float>
 *       - if the render parameter = true, play after set, else pause after set
 * - restartFieldAnimation - restarts/resumes video playback
 * - stopFieldAnimation - pauses video playback (immediate parameter ignored)
 *
 * To use this class, register the createjs object using the
 * static register function.  The video Component Parameter class name must
 * be specified as the second parameter ("myVideo" below). The created
 * HarmonicVideo object is returned from the register function.  If there is an
 * error, then 'undefined' is returned and an error will be logged.
 *
 * @example
 * // 'myVideo' is the name specified in the video Component Parameter class text box
 * // true = remove the poster attribute from the video element (default = true)
 * let harmonicVideo = HarmonicVideo.register(this, "myVideo", true);
 *
 * // The Harmonic Video widget will dispatch the VIDEO_ELEMENT_STATE_EVENT when the
 * // video element property is determined.  Ues isVideoElementValid() to
 * // determine if it is valid.
 * document.addEventListener(HarmonicVideo.VIDEO_ELEMENT_STATUS_EVENT, onVideoElementStatus.bind(this));
 *
 * function onVideoElementStatus(event) {
 *     hjs.info("onVideoElementStatus: valid: " + event.detail.isVideoElementValid() + " class: " + event.detail.class);
 * }
 */
class HarmonicVideo extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     * @param {String} className - as specified in the Adobe Animate video component
     * @param {Boolean} removePoster - if true, removes the poster attribute on the video element
     */
    constructor(widgetType, fieldNum, name, createjsObj, className, removePoster ) {
        super(widgetType, fieldNum, name, createjsObj, true);  // include in the template description

        hjs.debug("HarmonicVideo.constructor");

        /**
         * Name of the class parameter in the Adobe Animate video component parameters
         * @type {String}
         */
        this._class = className;

        /**
         * Flag indicating whether to remove the poster attribute on the video element.
         * If the poster is not removed and the image is blank, then the first frame
         * of video may be blank.
         * @type {Boolean}
         */
        this._removePoster = removePoster;

        /**
         * The video element - found in the DOM identified by the class name.
         * @type {Object}
         */
        this._video = null;

        /**
         * Flag indicating if the load processing is complete.
         * @type {Boolean}
         */
        this._loadComplete = false;

        /**
         * Counter used to determine if the load is taking too long.
         * @type {Number}
         */
        this._loadCounter = 0;

        /**
         * Tracks the state of the load processing.
         * @type {Number}
         */
        this._loadState = HarmonicVideo.STATE_WAITING_FOR_HAVE_ENOUGH_DATA;

        /**
         * Flag used to perform special processing on the first restartFieldAnimation.
         * @type {Boolean}
         */
        this._firstRestartFieldAnimation = true;

        requestAnimationFrame(() => {

            this._video = document.getElementById(this._class);
            if (this._video) {
                if (this._removePoster) {
                    this._video.removeAttribute("poster");
                }
            }
            else {
                hjs.error("Unable to locate the video element in the DOM: class: " + this._class);
            }

            this._dispatchEvent();
        });
    }

    /**
     * Registers a createjs object with the widget.  The className parameter must
     * match the Adobe Animate Component Parameter class value.  A new HarmonicVideo object
     * will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with this widget
     * @param {String} className - the Adobe Component Parameter class value
     * @param {Boolean} removePoster - if true, removes the poster attribute on the video element
     *
     * @return {Object} the newly created HarmonicVideo instance, or undefined if there is an error
     */
    static register(createjsObj, className, removePoster = true) {
        let result = HarmonicWidget.validate("HarmonicVideo", createjsObj, false);
        if (result) {
            return new HarmonicVideo(HarmonicTemplate.WIDGET_TYPE_VIDEO, result.fieldNum, result.name, createjsObj, className, removePoster);
        }
        else {
            return undefined;
        }
    }

    /**
     * Returns true if the video element is valid, else false
     * @returns {Boolean}
     */
    isVideoElementValid() {
        if (this._video) {
            return true;
        }
        else {
            return false;
        }
    }

    /**
     * Queried by the Harmonic Template to verify that the widget is loaded
     * before sending the load complete signal.
     *
     * Waits for the video element to report a readyState of HAVE_ENOUGH_DATA.
     * Once this is determined, an asynchronous play is invoked and the promise
     * is examined next.  If successful, video is paused and the loading process
     * is complete.
     *
     * @returns {Boolean}
     */
    isLoaded() {

        if (!this._video) {
            return false;
        }

        // Once the load processing is complete, exit without running the below logic.
        if (this._loadComplete) {
            return true;
        }

        this._loadCounter++;

        switch (this._loadState) {

            case HarmonicVideo.STATE_WAITING_FOR_HAVE_ENOUGH_DATA:

                // Wait for the readyState to be HAVE_ENOUGH_DATA before continuing.
                if (this._video.readyState == this._video.HAVE_ENOUGH_DATA) {
                    this._loadState = HarmonicVideo.STATE_WAITING_FOR_PLAY_PROMISE;

                    hjs.info("class: " + this._class + " event: HAVE_ENOUGH_DATA readyState: " + this._video.readyState + " counter: " + this._loadCounter);

                    // The play() is an asynchronous call that returns a promise.
                    // Update the state in the promise result.
                    let playPromise = this._video.play();
                    if (playPromise !== undefined) {
                        playPromise.then(function(result) {
                            this._video.pause();        // Pause video
                            this._loadState = HarmonicVideo.STATE_LOAD_COMPLETE;
                        }.bind(this),
                        function(err) {
                            hjs.error("Play was prevented: error: " + err);
                            this._loadState = HarmonicVideo.STATE_LOAD_COMPLETE;
                        }.bind(this));
                    }
                }
                break;

            case HarmonicVideo.STATE_WAITING_FOR_PLAY_PROMISE:
                // Waiting for the promise to return
                break;

            case HarmonicVideo.STATE_LOAD_COMPLETE:
                hjs.info("class: " + this._class + " event: LOAD_COMPLETE readyState: " + this._video.readyState + " counter: " + this._loadCounter);
                this._loadState = true;
                return true;

            default:
                return false;
        }
    }

    /**
     * Handles the updateTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    updateTextField(text, render) {
        if (!this._video) {
            hjs.error("HarmonicVideo.updateTextField failed: this._video is undefined");
            return;
        }

        if (text.startsWith("src:")) {
            this._video.setAttribute("src", text.slice(4));
        }
        else if (text.startsWith("currentTime:")) {
            let value = Number(text.slice(12));
            if (!Number.isNaN(value)) {
                this._video.currentTime = value;
            }
            else {
                hjs.error("HarmonicVideo.updateTextField failed: invalid number: " + value);
            }
        }
        else {
            hjs.error("HarmonicVideo.updateTextField failed: invalid command: " + text);
        }
    }

    /**
     * Handles the restartFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} fromBeginning - flag indicating if the widget should restart animation from the beginning
     */
    restartFieldAnimation(fromBeginning) {
        hjs.debug("restartFieldAnimation fromBeginning: " + fromBeginning);

        if (!this._video) {
            return;
        }

        if (fromBeginning) {
            if (this._firstRestartFieldAnimation) {
                this._firstRestartFieldAnimation = false;
                let autoplay = this._video.getAttribute("autoplay");

                // Only start playing if autoplay is specified
                if (autoplay != null) {
                    this._video.currentTime = 0.0;
                    this._video.play();
                }
            }
            else {
                this._video.pause();
                this._video.currentTime = 0.0;
                this._video.play();
            }
        }
        else {
            this._video.play();
        }
    }

    /**
     * Handles the stopFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} immediate - flag indicating if the widget should stop immediately or at the end of the animation
     */
    stopFieldAnimation(immediate) {
        hjs.debug("stopFieldAnimation immediate: " + immediate);

        if (!this._video) {
            return;
        }

        this._video.pause();
    }

    //--------------------------------------------------------------------------
    // Private Methods
    //--------------------------------------------------------------------------

    /**
     * Dispatches the VIDEO_ELEMENT_STATE_EVENT.
     */
    _dispatchEvent() {
        // Dispatch the video element status event.
        let event = new CustomEvent(HarmonicVideo.VIDEO_ELEMENT_STATUS_EVENT,
            {
                detail : this
            }
        );

        document.dispatchEvent(event);
    }

    //--------------------------------------------------------------------------
    // Properties
    //--------------------------------------------------------------------------
    /**
     * Gets the video element.
     */
    get video() {
        return this._video;
    }

    /**
     * Gets the class name specified at construction.
     */
    get class() {
        return this._class;
    }
}
