/**
 * The HarmonicTextField widget provides a dynamic text field that can
 * be controlled with the widget controls.
 *
 * This class implements the Harmonic TextField Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The HarmonicTextField widget requires that the registered createdjs
 * object have the following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 * - The createjs.MovieClip must contain a createjs.Text child named 'harmonicField'.
 *   This text object represents the dynamic text field.
 *
 * This class defines functions for:
 *
 * - updateTextField - updates and renders the dynamic text field
 * - appendTextField - appends text to the dynamic text field and renders
 * - renderField - renders the text field
 *
 * To use this class, simply register the createjs object using the
 * static register function.  The created HarmonicTextField object is
 * returned from the register function.  If there is a error, then
 * 'undefined' is returned and an error will be logged.
 *
 * @example
 * let textfield = HarmonicTextField.register(this);
 * textfield.updateTextField("updated text", true);
 */
class HarmonicTextField extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     */
    constructor(widgetType, fieldNum, name, createjsObj) {
        super(widgetType, fieldNum, name, createjsObj, true);  // include in the template description

        hjs.debug("HarmonicTextField.constructor");

        /**
         * The text to be rendered to the createjs.Text object.
         * @type {String}
         */
        this._text = this._createjsObj.harmonicField.text;
    }

    /**
     * Registers a createjs object with the widget.  A new
     * HarmonicTextField will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with this widget
     *
     * @return {Object} the newly created HarmonicTextField instance, or undefined if there is an error
     */
    static register(createjsObj) {
        let result = HarmonicWidget.validate("HarmonicTextField", createjsObj, true);
        if (result) {
            return new HarmonicTextField(HarmonicTemplate.WIDGET_TYPE_TEXT, result.fieldNum, result.name, createjsObj);
        }
        else {
            return undefined;
        }
    }

    /**
     * Handles the updateTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    updateTextField(text, render) {
        this._text = text

        if (render) {
            this._renderNow();
        }
    }

    /**
     * Handles the appendTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to append to the display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    appendTextField(text, render) {
        this._text += text;

        if (render) {
            this._renderNow();
        }
    }

    /**
     * Handles the renderField request from the Harmonic Template.
     */
    renderField() {
        this._renderNow();
    }

    //--------------------------------------------------------------------------
    // Private Methods
    //--------------------------------------------------------------------------

    /**
     * Renders the text to the createjs object and forces an update on the stage.
     */
    _renderNow() {
        this._createjsObj.harmonicField.text = this._text;
    }

    //--------------------------------------------------------------------------
    // Properties
    //--------------------------------------------------------------------------
    /**
     * Returns the string that is waiting to be displayed
     */
    get savedText() {
        return this._text;
    }

    /**
     * Returns the string that is currently being displayed
     */
    get displayedText() {
        return this._createjsObj.harmonicField.text;
    }
}
