/**
 * The HarmonicLabel widget provides a dynamic text field that can
 * be controlled with the widget controls.  Unlike the HarmonicTextField widget,
 * this widget uses the Label component so that customers can utilize CSS styling.
 * The Label component is rendered using HTML rendering, not CreateJS canvas
 * rendering.  This means that the label can only be placed above or below the
 * canvas.
 *
 * This class implements the Harmonic Label Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The HarmonicLabel widget requires that the registered createdjs object have the
 * following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 * - The createjs.MovieClip must contain a label component child named the same
 *   as the createjsObj (for example, field2_label).  This is naming convention
 *   is necessary to uniquely define the label in the DOM.
 *
 * Styling for the label should be done outside this class in a CSS file for the
 * template.  The class name for the label is defined by the 'class' property on
 * the label component.
 *
 * This class defines functions for:
 *
 * - updateTextField - updates and renders the dynamic text field
 * - appendTextField - appends text to the dynamic text field and renders
 * - renderField - renders the text field
 *
 * To use this class, simply register the createjs object using the
 * static register function.  The created HarmonicLabel object is
 * returned from the register function.  If there is a error, then
 * 'undefined' is returned and an error will be logged.
 *
 * @example
 * let label = HarmonicLabel.register(this, HarmonicLabel.LOCATION_ABOVE_CANVAS);
 *
 * // Calling label.updateTextField at this point will not work since additional
 * // Adobe generated code for inserting the label into the DOM must run.
 * // Use requestAnimationFrame to execute the updateTextField on the next
 * // available screen repaint.
 * requestAnimationFrame(() => {
 *   label.updateTextField("Testing", true);
 * });
 */
class HarmonicLabel extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     * @param {Number} location - location of label relative to canvas
     */
    constructor(widgetType, fieldNum, name, createjsObj, location) {
        super(widgetType, fieldNum, name, createjsObj, true);  // include in the template description

        hjs.debug("HarmonicLabel.constructor");

        /**
         * The location of the label relative to the canvas.
         * @type {Number} 0 = below, 1 = above
         */
        this._location = this._initLocation(location);

        /**
         * Flag indicating if the load processing is complete.
         * @type {Boolean}
         */
        this._loadComplete = false;

        /**
         * Object representing the label element in the DOM.
         */
        this._label = undefined;

        /**
         * The text to be rendered to the label object.
         * @type {String}
         */
        this._text = "";

        // The label should be present in the DOM by the next repaint.
        requestAnimationFrame(() => {
            let animation_container = document.getElementById("animation_container");
            let dom_overlay_container = document.getElementById("dom_overlay_container");
            let label = document.getElementById(this._createjsObj.name);

            if (!animation_container) {
                hjs.error("widget: " + this._createjsObj.name + " unable to locate 'animation_container' in the DOM");
                return;
            }

            if (!dom_overlay_container) {
                hjs.error("widget: " + this._createjsObj.name + "unable to locate 'dom_overlay_container' in the DOM");
                return;
            }

            if (!label) {
                hjs.error("widget: " + this._createjsObj.name + " - unable to locate the label element in the DOM");
                return;
            }

            this._label = label;
            this._text = this._label.innerHTML;

            // Position the label in the DOM
            if (this._location == HarmonicLabel.LOCATION_BELOW_CANVAS) {

                // Remove the label added by Adobe
                dom_overlay_container.removeChild(this._label);

                // Create the div below the canvas if it doesn't already exist.
                let node = document.getElementById("dom_overlay_container_below");
                if (!node) {
                    node = dom_overlay_container.cloneNode(false);
                    node.id = node.id + "_below";

                    node = animation_container.insertBefore(node, animation_container.firstChild);
                }

                node.appendChild(this._label);
            }

            this._loadComplete = true;
        });
    }

    /**
     * Registers a createjs object with the widget.  A new
     * HarmonicLabel will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with this widget
     * @param {Number} location - location of label relative to canvas
     *
     * @return {Object} the newly created HarmonicLabel instance, or undefined if there is an error
     */
    static register(createjsObj, location) {
        let result = HarmonicWidget.validate("HarmonicLabel", createjsObj, true);
        if (result) {
            return new HarmonicLabel(HarmonicTemplate.WIDGET_TYPE_TEXT, result.fieldNum, result.name, createjsObj, location);
        }
        else {
            return undefined;
        }
    }

    /**
     * Initializes the location of the label.
     *
     * @param {Number} location
     *
     * @return {Number} the verified location.
     */
    _initLocation(location) {
        if ((location >= 0) && (location < HarmonicLabel.LOCATION_MAX)) {
            return location;
        }
        else {
            hjs.error("widget: " + this._createjsObj.name + " - location is out of range: " + location.toString() + " defaulting to " + this.defaultLocation);
            return this.defaultLocation;
        }
    }

    /**
     * Queried by the Harmonic Template to verify that the widget is loaded
     * before sending the load complete signal.  The requestAnimationFrame
     * callback in the constructor will set the loadComplete flag to true.
     *
     * @returns {Boolean}
     */
    isLoaded() {
        return this._loadComplete;
    }

    /**
     * Handles the updateTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    updateTextField(text, render) {
        this._text = text

        if (render) {
            this._renderNow();
        }
    }

    /**
     * Handles the appendTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to append to the display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    appendTextField(text, render) {
        this._text += text;

        if (render) {
            this._renderNow();
        }
    }

    /**
     * Handles the renderField request from the Harmonic Template.
     */
    renderField() {
        this._renderNow();
    }

    //--------------------------------------------------------------------------
    // Private Methods
    //--------------------------------------------------------------------------

    /**
     * Renders the text to the label.
     */
    _renderNow() {
        if (this._label) {
            this._label.innerHTML = this._text;
        }
        else {
            hjs.error("widget: " + this._createjsObj.name + " - render failed: this._label is undefined");
        }
    }

    //--------------------------------------------------------------------------
    // Properties
    //--------------------------------------------------------------------------
    /**
     * Returns the default location.
     * @return {Number}
     */
    get defaultLocation() {
        return HarmonicLabel.LOCATION_ABOVE_CANVAS;
    }

    /**
     * Returns the string that is waiting to be displayed.
     */
    get savedText() {
        return this._text;
    }

    /**
     * Returns the string that is currently being displayed.
     */
    get displayedText() {
        if (this._label) {
            return this._label.innerHTML;
        }
        else {
            hjs.error("widget: " + this._createjsObj.name + " - this._label is undefined");
            return "";
        }
    }
}

//--------------------------------------------------------------------------
// Constants
//--------------------------------------------------------------------------
HarmonicLabel.LOCATION_BELOW_CANVAS = 0;
HarmonicLabel.LOCATION_ABOVE_CANVAS = 1;
HarmonicLabel.LOCATION_MAX = 2;
