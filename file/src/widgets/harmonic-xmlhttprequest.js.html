/**
 * The HarmonicXMLHttpRequest widget provides a mechanism to read data from
 * a webserver.  It contains a dynamic text field that can be used to load
 * the URL.
 *
 * This class implements the Harmonic XMLHttpRequest Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The XMLHttpRequest widget requires that the registered createdjs
 * object have the following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 * - The createjs.MovieClip must contain a createjs.Text child named 'harmonicField'.
 *   This text object represents the dynamic text field.
 *
 * This class defines a function for:
 *
 * - updateTextField - the text defines the URL to load from
 *
 * To use this class, simply register the createjs object using the
 * static register function.  The created HarmonicXMLHttpRequest object is
 * returned from the register function.  If there is a error, then
 * 'undefined' is returned and an error will be logged.
 *
 * The 'url' can be specified in the register function or it
 * can be set using the 'url' property on the object returned from
 * the call to register.
 *
 * After a load is request, the widget will dispatch the
 * HarmonicXMLHttpRequest.XML_HTTP_REQUEST_COMPLETE_EVENT event on the document or
 * the HarmonicXMLHttpRequest.XML_HTTP_REQUEST_TIMEOUT_EVENT event if the
 * asynchronous request has timed out
 *
 * @example
 * 1. When the createjs object is registered with the class in the first frame of the movie clip:
 * let xhr = HarmonicXMLHttpRequest.register(this, "test.xml");
 *
 * 2. Or set the 'url' property from the returned object:
 * let xhr = HarmonicXMLHttpRequest.register(this);
 * xhr.url = "test.xml";
 *
 * document.addEventListener(HarmonicXMLHttpRequest.XML_HTTP_REQUEST_COMPLETE_EVENT, (event) => {
 *     hjs.info("Your data is loaded: " + event.detail);
 * });
 *
 * document.addEventListener(HarmonicXMLHttpRequest.XML_HTTP_REQUEST_TIMEOUT_EVENT, (event) => {
 *     hjs.error("Your load timed out");
 * });
 *
 * xhr.load();
 */
class HarmonicXMLHttpRequest extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     */
    constructor(widgetType, fieldNum, name, createjsObj) {
        super(widgetType, fieldNum, name, createjsObj, true);  // include in the template description

        hjs.debug("HarmonicXMLHttpRequest.constructor");

        /**
         * The URL to send the request to.
         * @type {String}
         */
        this._url = "";

        /**
         * Flag to indicate if the request should be asynchronous or synchronous.
         * @type {Boolean}
         */
        this._async = true;

        /**
         * The timeout property of the XMLHttpRequest.stringify
         * @type {Number}
         */
        this._timeout = 0;

        /**
         * Flag that indicates if the load should occur when the URL changes.
         * @type {Boolean}
         */
        this._loadWhenUrlChanges = true;

        /**
         * The XMLHttpRequest object.
         * @type {Object}
         */
        this._xhttp = undefined;
    }

    /**
     * Registers a createjs object with the widget.  A new
     * HarmonicXMLHttpRequest will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with this widget
     * @param {String} url - the url to send the request to
     *
     * @return {Object} the newly created HarmonicXMLHttpRequest instance, or undefined if there is an error
     */
    static register(createjsObj, url) {
        let result = HarmonicWidget.validate("HarmonicXMLHttpRequest", createjsObj, true);
        if (result) {
            return new HarmonicXMLHttpRequest(HarmonicTemplate.WIDGET_TYPE_TEXT, result.fieldNum, result.name, createjsObj, url);
        }
        else {
            return undefined;
        }
    }

    /**
     * Handles the updateTextField request from the Harmonic Template.
     *
     * @param {String} text - the text to display
     * @param {Boolean} render - flag indicating if the field should be rendered
     */
    updateTextField(text, render) {
        this._url = text;
        if (this._loadWhenUrlChanges) {
            this.load();
        }
    }

    /**
     * Loads the contents from the url.  The XML_HTTP_REQUEST_COMPLETE_EVENT will
     * be dispatched when completed.
     *
     * @emits {HarmonicXMLHttpRequest.XML_HTTP_REQUEST_COMPLETE_EVENT} is dispatched when the request completes
     */
    load() {
        if (this._url) {
            this._xhttp = new XMLHttpRequest();
            this._xhttp.open("GET", this._url, this._async);

            if (this._async) {
                this._xhttp.timeout = this._timeout;
            }

            this._xhttp.onreadystatechange = function() {
                // hjs.debug("readyState: " + this._xhttp.readyState + " status: " + this._xhttp.status + " statusText: " + this._xhttp.statusText);
                if (this._xhttp.readyState == XMLHttpRequest.DONE) {
                    let obj = {};
                    if (this._xhttp.status == 200) {
                        obj.status = true;
                        obj.httpStatusCode = this._xhttp.status;
                        obj.text = this._xhttp.responseText;

                        // hjs.debug("HarmonicXMLHttpRequest.load success: url " + this._url + " text: " + obj.text);
                    }
                    else {
                        obj.status = false;
                        obj.httpStatusCode = this._xhttp.status;
                        obj.text = this._xhttp.statusText;

                        hjs.error("HarmonicXMLHttpRequest.load failed: url " + this._url + " statusText: " + obj.text);
                    }

                    let event = new CustomEvent(HarmonicXMLHttpRequest.XML_HTTP_REQUEST_COMPLETE_EVENT,
                        {
                            detail : obj
                        }
                    );

                    document.dispatchEvent(event);
                }
            }.bind(this);

            this._xhttp.ontimeout = function() {
                let event = new CustomEvent(HarmonicXMLHttpRequest.XML_HTTP_REQUEST_TIMEOUT_EVENT);
                document.dispatchEvent(event);
            }

            this._xhttp.send();
        }
        else {
            hjs.error("HarmonicXMLHttpRequest.load: invalid url: " + this._url);
        }
    }

    //--------------------------------------------------------------------------
    // Properties
    //--------------------------------------------------------------------------

    /**
     * Sets the URL.
     * @param {String} value - the URL to send the request to.
     */
    set url(value) {
        this._url = value;
    }

    /**
     * Gets the URL.
     */
    get url() {
        return this._url;
    }

    /**
     * Sets the asynchronous flag on the request.
     * @param {Boolean} value - true = asynchronous, false = synchronous (deprecated)
     */
    set async(value) {
        this._async = value;
    }

    /**
     * Gets the asynchronous flag.
     */
    get async() {
        return this._async;
    }

    /**
     * Sets the timeout (milliseconds) on the request.
     * @param {Number} value - 0 = no timeout, > 0 = milliseconds
     */
    set timeout(value) {
        this._timeout = value;
    }

    /**
     * Gets the timeout (milliseconds)
     */
    get timeout() {
        return this._timeout;
    }

    /**
     * Sets the flag that indicates to load when the URL changes.
     * @param {Boolean} value - true = load when URL changes, false otherwise
     */
    set loadWhenUrlChanges(value) {
        this._loadWhenUrlChanges = value;
    }

    /**
     * Gets the value of that indicates whether or not to load when the URL changes.
     */
    get loadWhenUrlChanges() {
        return this._loadWhenUrlChanges;
    }
}

//------------------------------------------------------------------------------
// Events
//------------------------------------------------------------------------------
HarmonicXMLHttpRequest.XML_HTTP_REQUEST_COMPLETE_EVENT = "XMLHttpRequestCompleteEvent";
HarmonicXMLHttpRequest.XML_HTTP_REQUEST_TIMEOUT_EVENT = "XMLHttpRequestTimeoutEvent";
