/**
 * The HarmonicMovieClip widget provides an easy way to control a
 * createjs.MovieClip with the widget controls.
 *
 * This class implements the Harmonic HarmonicMovieClip Widget.  It derives from
 * the {@link HarmonicWidget} class and implements necessary functions to meet the
 * needs of this widget.
 *
 * The HarmonicMovieClip widget requires that the registered createdjs
 * object have the following structure:
 *
 * - The createjsObj must be a createjs.MovieClip named correctly (reference
 *   {@link HarmonicWidget}).
 *
 * This class defines functions for:
 *
 * - restartFieldAnimation - restarts the animation
 * - stopFieldAnimation - stops the animation
 *
 * To use this class, simply register the createjs object using the
 * static register function.  The created HarmonicMovieClip object is
 * returned from the register function.  If there is a error, then
 * 'undefined' is returned and an error will be logged.
 *
 * @example
 * let movieClip = HarmonicMovieClip.register(this);
 * movieClip.stopFieldAnimation(true);
 */
class HarmonicMovieClip extends HarmonicWidget {

    /**
     * Constructor
     * - initializes all member variables
     * - adds a listener to the 'tick' event so that the count can be updated
     *
     * @listens {'tick'} to detect if the timeline needs to stop at the last frame
     *
     * @param {String} widgetType - reference HarmonicTemplate.WIDGET_TYPE_<type>
     * @param {Number} fieldNum - field number used to uniquely identify the widget
     * @param {String} name - the name of the widget
     * @param {Object} createjsObj - the createjs object associated with this widget
     */
    constructor(widgetType, fieldNum, name, createjsObj) {
        super(widgetType, fieldNum, name, createjsObj, false);  // don't include in the template description

        hjs.debug("HarmonicMovieClip.constructor");

        /**
         * Stop the MovieClip at the end of the timeline.
         * @type {Boolean}
         */
        this._stopAtEndRequested = false;

        // Listen for the 'tick' event on the object
        createjsObj.addEventListener("tick", (event) => {
            if (this._stopAtEndRequested) {
                if (this._createjsObj.currentFrame == (this._createjsObj.totalFrames - 1)) {
                    this.createjsObj.stop();
                }
            }
        });
    }

    /**
     * Registers a createjs object with the widget.  A new
     * HarmonicMovieClip will be created in this function.
     *
     * @param {Object} createjsObj - the createjs object to associate with this widget
     *
     * @return {Object} the newly created HarmonicMovieClip instance, or undefined if there is an error
     */
    static register(createjsObj) {
        let result = HarmonicWidget.validate("HarmonicMovieClip", createjsObj, false);
        if (result) {
            return new HarmonicMovieClip(HarmonicTemplate.WIDGET_TYPE_MOVIE_CLIP, result.fieldNum, result.name, createjsObj);
        }
        else {
            return undefined;
        }
    }

    /**
     * Handles the restartFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} fromBeginning - flag indicating if the widget should restart animation from the beginning
     */
    restartFieldAnimation(fromBeginning) {
        this._stopAtEndRequested = false;

        if (fromBeginning) {
            this._createjsObj.gotoAndPlay(0);
        }
        else {
            this._createjsObj.play();
        }
    }

    /**
     * Handles the stopFieldAnimation request from the Harmonic Template.
     *
     * @param {Boolean} immediate - flag indicating if the widget should stop immediately or at the end of the animation
     */
    stopFieldAnimation(immediate) {
        if (immediate) {
            this._createjsObj.stop();
        }
        else {
            this._stopAtEndRequested = true;
        }
    }
}
