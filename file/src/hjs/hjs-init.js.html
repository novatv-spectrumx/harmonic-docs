    /**
 * File: hjs-init.js
 * 
 * Copyright (c) 2018 Harmonic, Inc.
 * 
 * Handles construction and initialization of HJS, Harmonic Template,
 * and Harmonic Widgets.
 * 
 */

//------------------------------------------------------------------------------
// HJS Configuration
//------------------------------------------------------------------------------

/**
 * Global object that represents the HJS configuration.  
 * @type {Object}
 */
let hjsConfig = {
    severity: HJS.INFO,                    // Initial log severity
    remoteWebSocket : {                    // Connect to a remote websocket
        enable: false,                     // enable for connection
        ipAddress: "",                     // websocket server ip address
        port: 9500                         // websocket server port
    }
};

//------------------------------------------------------------------------------
// Harmonic JavaScript (aka HJS)
//------------------------------------------------------------------------------

/**
 * Global object that represents the HJS.  The HJS provides the interface to
 * and from the HTML renderer.
 * @type {HJS} 
 */
let hjs = new HJS(hjsConfig);

//------------------------------------------------------------------------------
// Harmonic Template and Widgets
//------------------------------------------------------------------------------

/**
 * Global object that provides the Harmonic Template functionality.  
 * @type {HarmonicTemplate}
 */
let harmonicTemplate = new HarmonicTemplate();

/**
 * Global object that contains the 2 Harmonic DVE objects.
 * @type {HarmonicDVE[]}
 */
let harmonicDVE = [ new HarmonicDVE(0), new HarmonicDVE(1) ];

/**
 * Global object that represents DVE 0 (standard/UHD channel).
 * @type {HarmonicDVE}
 */
let dve = harmonicDVE[0]; 

/**
 * Global object that represents DVE 0 (enhanced channel).
 * @type {HarmonicDVE}
 */
let dve0 = dve;

/**
 * Global object that represents DVE 1 (enhanced channel).
 * @type {HarmonicDVE}
 */
let dve1 = harmonicDVE[1];

/**
 * Global object that contains the 2 Harmonic Live AVO objects.
 * @type {HarmonicLiveAVO[]}
 */
let harmonicLiveAVO = [ new HarmonicLiveAVO(0), new HarmonicLiveAVO(1) ];

/**
 * Global object that represents Live AVO unit 0 (standard/UHD channel).
 * @type {HarmonicDVE}
 */
let liveAVO = harmonicLiveAVO[0]; 

/**
 * Global object that represents Live AVO unit 0 (enhanced channel).
 * @type {HarmonicDVE}
 */
let liveAVO0 = liveAVO;

/**
 * Global object that represents Live AVO unit 1 (enhanced channel).
 * @type {HarmonicDVE}
 */
let liveAVO1 = harmonicLiveAVO[1];

// Initialize HJS after creating the Harmonic Template and Widgets
hjs.init();
